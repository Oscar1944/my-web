  //getReference() 函式內可輸入相關之參數(資料庫內的位置參照名稱)，用來指定參考至資料庫的特定位置

  //要對ListView載入資料，必須使用Adapter物件，專題使用的ArrayAdapter只是其中一種

  //ListView的Layout方式多種，simple_list_item_1只是其中一種，這些Layout方式內建在Android內，是專門為ListView的item所使用的，因此
須從android.R.layout中呼叫使用

  //為ListView設定上Adapter

  //addValueEventListener是FireBase中，可以進入getReference()指向的資料庫位置獲取資料的方式之一
ref_contact.addValueEventListener(new ValueEventListener() {
  @Override
  public void onDataChange(@NonNull DataSnapshot dataSnapshot) { //DataSnapshot物件可以理解為，
    從資料庫傳回的資料集合之物件，資料內容就是先前指定資料庫位置參照的資料內容
      adapter.clear(); //對adapter初始化
      String name;
      String location;

      String x =dataSnapshot.child("t").getValue().toString(); //從資料庫內取得資料"t"，"t"是資料庫內紀錄資料總數的欄位
      int t =Integer.parseInt(x);
      String s;
      for(int i =0;i < t;i++){  //用迴圈逐一取出對應順序的景點名稱及地點資料
          s =String.valueOf(i);
          name =dataSnapshot.child(s).child("0").getValue().toString();
          location =dataSnapshot.child(s).child("1").getValue().toString();

          adapter.add(name+"\n"+location);  //將取出的資料加入adapter內
      }
  }
